108 //---------------------------------------------------------------------------
109 bool Semaphore::Post()
110 {
111  KERNEL_ASSERT(IsInitialized());
112 
113  auto bThreadWake = false;
114  auto bBail = false;
115  // Increment the semaphore count - we can mess with threads so ensure this
116  // is in a critical section. We don't just disable the scheudler since
117  // we want to be able to do this from within an interrupt context as well.
118  CS_ENTER();
119 
120  // If nothing is waiting for the semaphore
121  if (m_clBlockList.GetHead() == nullptr) {
122  // Check so see if we've reached the maximum value in the semaphore
123  if (m_u16Value < m_u16MaxValue) {
124  // Increment the count value
125  m_u16Value++;
126  } else {
127  // Maximum value has been reached, bail out.
128  bBail = true;
129  }
130  } else {
131  // Otherwise, there are threads waiting for the semaphore to be
132  // posted, so wake the next one (highest priority goes first).
133  bThreadWake = (WakeNext() != 0u);
134  }
135  CS_EXIT();
136 
137  // If we weren't able to increment the semaphore count, fail out.
138  if (bBail) {
139  return false;
140  }
141 
142  // if bThreadWake was set, it means that a higher-priority thread was
143  // woken. Trigger a context switch to ensure that this thread gets
144  // to execute next.
145  if (bThreadWake) {
146  Thread::Yield();
147  }
148  return true;
149 }

152 bool Semaphore::Pend_i(uint32_t u32WaitTimeMS_)
153 {
154  KERNEL_ASSERT(IsInitialized());
155 
156  Timer clSemTimer;
157  auto bUseTimer = false;
158 
159  // Once again, messing with thread data - ensure
160  // we're doing all of these operations from within a thread-safe context.
161  CS_ENTER();
162 
163  // Check to see if we need to take any action based on the semaphore count
164  if (m_u16Value != 0) {
165  // The semaphore count is non-zero, we can just decrement the count
166  // and go along our merry way.
167  m_u16Value--;
168  } else {
169  // The semaphore count is zero - we need to block the current thread
170  // and wait until the semaphore is posted from elsewhere.
171  if (u32WaitTimeMS_ != 0u) {
172  g_pclCurrent->SetExpired(false);
173  clSemTimer.Init();
174  clSemTimer.Start(false, u32WaitTimeMS_, TimedSemaphore_Callback, this);
175  bUseTimer = true;
176  }
177  BlockPriority(g_pclCurrent);
178 
179  // Switch Threads immediately
180  Thread::Yield();
181  }
182 
183  CS_EXIT();
184 
185  if (bUseTimer) {
186  clSemTimer.Stop();
187  return (static_cast<int>(g_pclCurrent->GetExpired()) == 0);
188  }
189  return true;
190 }
