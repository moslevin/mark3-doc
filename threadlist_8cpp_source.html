53 void ThreadList::AddPriority(LinkListNode* node_)
54 {
55  KERNEL_ASSERT(node_ != nullptr);
56  auto* pclCurr = static_cast<Thread*>(GetHead());
57  if (pclCurr == nullptr) {
58  Add(node_);
59  return;
60  }
61  auto uXHeadPri = pclCurr->GetCurPriority();
62 
63  auto* pclTail = static_cast<Thread*>(GetTail());
64  auto* pclNode = static_cast<Thread*>(node_);
65 
66  // Set the threadlist's priority level, flag pointer, and then add the
67  // thread to the threadlist
68  auto uXPriority = pclNode->GetCurPriority();
69  do {
70  if (uXPriority > pclCurr->GetCurPriority()) {
71  break;
72  }
73  pclCurr = static_cast<Thread*>(pclCurr->GetNext());
74  } while (pclCurr != pclTail);
75 
76  // Insert pclNode before pclCurr in the linked list.
77  InsertNodeBefore(pclNode, pclCurr);
78 
79  // If the priority is greater than current head, reset
80  // the head pointer.
81  if (uXPriority > uXHeadPri) {
82  m_pclHead = pclNode;
83  m_pclTail = m_pclHead->prev;
84  } else if (pclNode->GetNext() == m_pclHead) {
85  m_pclTail = pclNode;
86  }
87 }
