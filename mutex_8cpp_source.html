109 bool Mutex::Claim_i(uint32_t u32WaitTimeMS_)
110 {
111  KERNEL_ASSERT(IsInitialized());
112 
113  Timer clTimer;
114  auto bUseTimer = false;
115 
116  // Disable the scheduler while claiming the mutex - we're dealing with all
117  // sorts of private thread data, can't have a thread switch while messing
118  // with internal data structures.
119  Scheduler::SetScheduler(false);
120 
121  // Check to see if the mutex is claimed or not
122  if (static_cast<int>(m_bReady) != 0) {
123  // Mutex isn't claimed, claim it.
124  m_bReady = false;
125  m_u8Recurse = 0;
126  m_u8MaxPri = g_pclCurrent->GetPriority();
127  m_pclOwner = g_pclCurrent;
128 
129  Scheduler::SetScheduler(true);
130  return true;
131  }
132 
133  // If the mutex is already claimed, check to see if this is the owner thread,
134  // since we allow the mutex to be claimed recursively.
135  if (m_bRecursive && (g_pclCurrent == m_pclOwner)) {
136  // Ensure that we haven't exceeded the maximum recursive-lock count
137  KERNEL_ASSERT((m_u8Recurse < 255));
138  m_u8Recurse++;
139 
140  // Increment the lock count and bail
141  Scheduler::SetScheduler(true);
142  return true;
143  }
144 
145  // The mutex is claimed already - we have to block now. Move the
146  // current thread to the list of threads waiting on the mutex.
147  if (u32WaitTimeMS_ != 0u) {
148  g_pclCurrent->SetExpired(false);
149  clTimer.Init();
150  clTimer.Start(false, u32WaitTimeMS_, (TimerCallback)TimedMutex_Callback, this);
151  bUseTimer = true;
152  }
153  BlockPriority(g_pclCurrent);
154 
155  // Check if priority inheritence is necessary. We do this in order
156  // to ensure that we don't end up with priority inversions in case
157  // multiple threads are waiting on the same resource.
158  if (m_u8MaxPri <= g_pclCurrent->GetPriority()) {
159  m_u8MaxPri = g_pclCurrent->GetPriority();
160 
161  auto* pclTemp = static_cast<Thread*>(m_clBlockList.GetHead());
162  while (pclTemp != nullptr) {
163  pclTemp->InheritPriority(m_u8MaxPri);
164  if (pclTemp == static_cast<Thread*>(m_clBlockList.GetTail())) {
165  break;
166  }
167  pclTemp = static_cast<Thread*>(pclTemp->GetNext());
168  }
169  m_pclOwner->InheritPriority(m_u8MaxPri);
170  }
171 
172  // Done with thread data -reenable the scheduler
173  Scheduler::SetScheduler(true);
174 
175  // Switch threads if this thread acquired the mutex
176  Thread::Yield();
177 
178  if (bUseTimer) {
179  clTimer.Stop();
180  return (static_cast<int>(g_pclCurrent->GetExpired()) == 0);
181  }
182  return true;
183 }

198 void Mutex::Release()
199 {
200  KERNEL_ASSERT(IsInitialized());
201 
202  auto bSchedule = false;
203 
204  // Disable the scheduler while we deal with internal data structures.
205  Scheduler::SetScheduler(false);
206 
207  // This thread had better be the one that owns the mutex currently...
208  KERNEL_ASSERT((g_pclCurrent == m_pclOwner));
209 
210  // If the owner had claimed the lock multiple times, decrease the lock
211  // count and return immediately.
212  if (m_bRecursive && (m_u8Recurse != 0u)) {
213  m_u8Recurse--;
214  Scheduler::SetScheduler(true);
215  return;
216  }
217 
218  // Restore the thread's original priority
219  if (g_pclCurrent->GetCurPriority() != g_pclCurrent->GetPriority()) {
220  g_pclCurrent->SetPriority(g_pclCurrent->GetPriority());
221 
222  // In this case, we want to reschedule
223  bSchedule = true;
224  }
225 
226  // No threads are waiting on this semaphore?
227  if (m_clBlockList.GetHead() == nullptr) {
228  // Re-initialize the mutex to its default values
229  m_bReady = true;
230  m_u8MaxPri = 0;
231  m_pclOwner = nullptr;
232  } else {
233  // Wake the highest priority Thread pending on the mutex
234  if (WakeNext() != 0u) {
235  // Switch threads if it's higher or equal priority than the current thread
236  bSchedule = true;
237  }
238  }
239 
240  // Must enable the scheduler again in order to switch threads.
241  Scheduler::SetScheduler(true);
242  if (bSchedule) {
243  // Switch threads if a higher-priority thread was woken
244  Thread::Yield();
245  }
246 }
