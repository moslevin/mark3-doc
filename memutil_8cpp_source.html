496 uint8_t MemUtil::Tokenize(const char* szBuffer_, Token_t* pastTokens_, uint8_t u8MaxTokens_)
497 {
498  uint8_t u8CurrArg = 0;
499  uint8_t u8LastArg = 0;
500  uint8_t i = 0;
501 
502  bool bEscape = false;
503 
504  KERNEL_ASSERT(szBuffer_);
505  KERNEL_ASSERT(pastTokens_);
506 
507  while (szBuffer_[i] != 0) {
508  //-- Handle unescaped quotes
509  if (szBuffer_[i] == '\"') {
510  bEscape = !bEscape;
511  i++;
512  continue;
513  }
514 
515  //-- Handle all escaped chars - by ignoring them
516  if (szBuffer_[i] == '\\') {
517  i++;
518  if (szBuffer_[i] != 0) {
519  i++;
520  }
521  continue;
522  }
523 
524  //-- Process chars based on current escape characters
525  if (bEscape) {
526  // Everything within the quote is treated as literal, but escaped chars are still treated the same
527  i++;
528  continue;
529  }
530 
531  //-- Non-escaped case
532  if (szBuffer_[i] != ' ') {
533  i++;
534  continue;
535  }
536 
537  pastTokens_[u8CurrArg].pcToken = &(szBuffer_[u8LastArg]);
538  pastTokens_[u8CurrArg].u8Len = i - u8LastArg;
539  u8CurrArg++;
540  if (u8CurrArg >= u8MaxTokens_) {
541  return u8MaxTokens_;
542  }
543 
544  i++;
545  while (szBuffer_[i] == ' ') { i++; }
546 
547  u8LastArg = i;
548  }
549  if ((i != 0u) && (szBuffer_[i] == 0) && ((i - u8LastArg) != 0)) {
550  pastTokens_[u8CurrArg].pcToken = &(szBuffer_[u8LastArg]);
551  pastTokens_[u8CurrArg].u8Len = i - u8LastArg;
552  u8CurrArg++;
553  }
554  return u8CurrArg;
