83 uint16_t EventFlag::Wait_i(uint16_t u16Mask_, EventFlagOperation eMode_, uint32_t u32TimeMS_)
84 {
85  KERNEL_ASSERT(eMode_ <= EventFlagOperation::Pending_Unblock);
86  KERNEL_ASSERT(IsInitialized());
87 
88  auto bThreadYield = false;
89  auto bMatch = false;
90 
91  Timer clEventTimer;
92  auto bUseTimer = false;
93 
94  // Ensure we're operating in a critical section while we determine
95  // whether or not we need to block the current thread on this object.
96  CS_ENTER();
97 
98  // Check to see whether or not the current mask matches any of the
99  // desired bits.
100  g_pclCurrent->SetEventFlagMask(u16Mask_);
101 
102  if ((eMode_ == EventFlagOperation::All_Set) || (eMode_ == EventFlagOperation::All_Clear)) {
103  // Check to see if the flags in their current state match all of
104  // the set flags in the event flag group, with this mask.
105  if ((m_u16SetMask & u16Mask_) == u16Mask_) {
106  bMatch = true;
107  g_pclCurrent->SetEventFlagMask(u16Mask_);
108 
109  if (EventFlagOperation::All_Clear == eMode_) {
110  m_u16SetMask &= ~u16Mask_;
111  g_pclCurrent->SetExpired(false);
112  }
113  }
114  } else if ((eMode_ == EventFlagOperation::Any_Set) || (eMode_ == EventFlagOperation::Any_Clear)) {
115  // Check to see if the existing flags match any of the set flags in
116  // the event flag group with this mask
117  if ((m_u16SetMask & u16Mask_) != 0) {
118  bMatch = true;
119  g_pclCurrent->SetEventFlagMask(m_u16SetMask & u16Mask_);
120 
121  if (EventFlagOperation::Any_Clear == eMode_) {
122  m_u16SetMask &= ~u16Mask_;
123  g_pclCurrent->SetExpired(false);
124  }
125  }
126  }
127 
128  // We're unable to match this pattern as-is, so we must block.
129  if (!bMatch) {
130  // Reset the current thread's event flag mask & mode
131  g_pclCurrent->SetEventFlagMask(u16Mask_);
132  g_pclCurrent->SetEventFlagMode(eMode_);
133 
134  if (u32TimeMS_ != 0u) {
135  g_pclCurrent->SetExpired(false);
136  clEventTimer.Init();
137  clEventTimer.Start(false, u32TimeMS_, TimedEventFlag_Callback, this);
138  bUseTimer = true;
139  }
140 
141  // Add the thread to the object's block-list.
142  BlockPriority(g_pclCurrent);
143 
144  // Trigger that
145  bThreadYield = true;
146  }
147 
148  // If bThreadYield is set, it means that we've blocked the current thread,
149  // and must therefore rerun the scheduler to determine what thread to
150  // switch to.
151  if (bThreadYield) {
152  // Switch threads immediately
153  Thread::Yield();
154  }
155 
156  // Exit the critical section and return back to normal execution
157  CS_EXIT();
158 
163  if (bUseTimer && bThreadYield) {
164  clEventTimer.Stop();
165  }
166 
167  return g_pclCurrent->GetEventFlagMask();
168 }

184 //---------------------------------------------------------------------------
185 void EventFlag::Set(uint16_t u16Mask_)
186 {
187  KERNEL_ASSERT(IsInitialized());
188 
189  Thread* pclPrev;
190  Thread* pclCurrent;
191  auto bReschedule = false;
192  uint16_t u16NewMask;
193 
194  CS_ENTER();
195 
196  // Walk through the whole block list, checking to see whether or not
197  // the current flag set now matches any/all of the masks and modes of
198  // the threads involved.
199 
200  m_u16SetMask |= u16Mask_;
201  u16NewMask = m_u16SetMask;
202 
203  // Start at the head of the list, and iterate through until we hit the
204  // "head" element in the list again. Ensure that we handle the case where
205  // we remove the first or last elements in the list, or if there's only
206  // one element in the list.
207  pclCurrent = static_cast<Thread*>(m_clBlockList.GetHead());
208 
209  // Do nothing when there are no objects blocking.
210  if (pclCurrent != nullptr) {
211  // First loop - process every thread in the block-list and check to
212  // see whether or not the current flags match the event-flag conditions
213  // on the thread.
214  do {
215  pclPrev = pclCurrent;
216  pclCurrent = static_cast<Thread*>(pclCurrent->GetNext());
217 
218  // Read the thread's event mask/mode
219  uint16_t u16ThreadMask = pclPrev->GetEventFlagMask();
220  auto eThreadMode = pclPrev->GetEventFlagMode();
221 
222  // For the "any" mode - unblock the blocked threads if one or more bits
223  // in the thread's bitmask match the object's bitmask
224  if ((EventFlagOperation::Any_Set == eThreadMode) || (EventFlagOperation::Any_Clear == eThreadMode)) {
225  if ((u16ThreadMask & m_u16SetMask) != 0) {
226  pclPrev->SetEventFlagMode(EventFlagOperation::Pending_Unblock);
227  pclPrev->SetEventFlagMask(m_u16SetMask & u16ThreadMask);
228  bReschedule = true;
229 
230  // If the "clear" variant is set, then clear the bits in the mask
231  // that caused the thread to unblock.
232  if (EventFlagOperation::Any_Clear == eThreadMode) {
233  u16NewMask &= ~(u16ThreadMask & u16Mask_);
234  }
235  }
236  }
237  // For the "all" mode, every set bit in the thread's requested bitmask must
238  // match the object's flag mask.
239  else if ((EventFlagOperation::All_Set == eThreadMode) || (EventFlagOperation::All_Clear == eThreadMode)) {
240  if ((u16ThreadMask & m_u16SetMask) == u16ThreadMask) {
241  pclPrev->SetEventFlagMode(EventFlagOperation::Pending_Unblock);
242  pclPrev->SetEventFlagMask(u16ThreadMask);
243  bReschedule = true;
244 
245  // If the "clear" variant is set, then clear the bits in the mask
246  // that caused the thread to unblock.
247  if (EventFlagOperation::All_Clear == eThreadMode) {
248  u16NewMask &= ~(u16ThreadMask & u16Mask_);
249  }
250  }
251  }
252  }
253  // To keep looping, ensure that there's something in the list, and
254  // that the next item isn't the head of the list.
255  while (pclPrev != m_clBlockList.GetTail());
256 
257  // Second loop - go through and unblock all of the threads that
258  // were tagged for unblocking.
259  pclCurrent = static_cast<Thread*>(m_clBlockList.GetHead());
260  auto bIsTail = false;
261  do {
262  pclPrev = pclCurrent;
263  pclCurrent = static_cast<Thread*>(pclCurrent->GetNext());
264 
265  // Check to see if this is the condition to terminate the loop
266  if (pclPrev == m_clBlockList.GetTail()) {
267  bIsTail = true;
268  }
269 
270  // If the first pass indicated that this thread should be
271  // unblocked, then unblock the thread
272  if (pclPrev->GetEventFlagMode() == EventFlagOperation::Pending_Unblock) {
273  UnBlock(pclPrev);
274  }
275  } while (!bIsTail);
276  }
277 
278  // If we awoke any threads, re-run the scheduler
279  if (bReschedule) {
280  Thread::Yield();
281  }
282 
283  // Update the bitmask based on any "clear" operations performed along
284  // the way
285  m_u16SetMask = u16NewMask;
286 
287  // Restore interrupts - will potentially cause a context switch if a
288  // thread is unblocked.
289  CS_EXIT();
290 }
