64 
65  auto* pclCurr = static_cast<Timer*>(GetHead());
66  // Subtract the elapsed time interval from each active timer.
67  while (pclCurr != 0) {
68  auto* pclNext = static_cast<Timer*>(pclCurr->GetNext());
69 
70  // Active timers only...
71  if ((pclCurr->m_u8Flags & TIMERLIST_FLAG_ACTIVE) != 0) {
72  pclCurr->m_u32TimeLeft--;
73  if (0 == pclCurr->m_u32TimeLeft) {
74  // Expired -- run the callback. these callbacks must be very fast...
75  if (pclCurr->m_pfCallback != nullptr) {
76  pclCurr->m_pfCallback(pclCurr->m_pclOwner, pclCurr->m_pvData);
77  }
78 
79  if ((pclCurr->m_u8Flags & TIMERLIST_FLAG_ONE_SHOT) != 0) {
80  // If this was a one-shot timer, deactivate the timer + remove
81  pclCurr->m_u8Flags |= TIMERLIST_FLAG_EXPIRED;
82  pclCurr->m_u8Flags &= ~TIMERLIST_FLAG_ACTIVE;
83  Remove(pclCurr);
84  } else {
