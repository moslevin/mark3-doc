38 void ThreadPort::InitStack(Thread* pclThread_)
39 {
40  // Initialize the stack for a Thread
41  uint16_t u16Addr;
42  uint8_t* pu8Stack;
43  uint16_t i;
44 
45  // Get the address of the thread's entry function
46  u16Addr = (uint16_t)(pclThread_->m_pfEntryPoint);
47 
48  // Start by finding the bottom of the stack
49  pu8Stack = (uint8_t*)pclThread_->m_pwStackTop;
50 
51 #if KERNEL_STACK_CHECK
52  // clear the stack, and initialize it to a known-default value (easier
53  // to debug when things go sour with stack corruption or overflow)
54  for (i = 0; i < pclThread_->m_u16StackSize; i++) { pclThread_->m_pwStack[i] = 0xFF; }
55 #endif // #if KERNEL_STACK_CHECK
56 
57  // Our context starts with the entry function
58  PUSH_TO_STACK(pu8Stack, (uint8_t)(u16Addr & 0x00FF));
59  PUSH_TO_STACK(pu8Stack, (uint8_t)((u16Addr >> 8) & 0x00FF));
60 
61  // R0
62  PUSH_TO_STACK(pu8Stack, 0x00); // R0
63 
64  // Push status register and R1 (which is used as a constant zero)
65  PUSH_TO_STACK(pu8Stack, 0x80); // SR
66  PUSH_TO_STACK(pu8Stack, 0x00); // R1
67 
68  // Push other registers
69  for (i = 2; i <= 23; i++) // R2-R23
70  {
71  PUSH_TO_STACK(pu8Stack, i);
72  }
73 
74  // Assume that the argument is the only stack variable
75  PUSH_TO_STACK(pu8Stack, (uint8_t)(((uint16_t)(pclThread_->m_pvArg)) & 0x00FF)); // R24
76  PUSH_TO_STACK(pu8Stack, (uint8_t)((((uint16_t)(pclThread_->m_pvArg)) >> 8) & 0x00FF)); // R25
77 
78  // Push the rest of the registers in the context
79  for (i = 26; i <= 31; i++) { PUSH_TO_STACK(pu8Stack, i); }
80 
81  PUSH_TO_STACK(pu8Stack, 0x00); // RAMPZ
82  // Set the top o' the stack.
83  pclThread_->m_pwStackTop = (uint8_t*)pu8Stack;
84 
85  // That's it! the thread is ready to run now.
86 }

95 void ThreadPort::StartThreads()
96 {
97  KernelSWI::Config(); // configure the task switch SWI
98  KernelTimer::Config(); // configure the kernel timer
99 
100  // Tell the kernel that we're ready to start scheduling threads
101  // for the first time.
102  Kernel::CompleteStart();
103 
104  Scheduler::SetScheduler(1); // enable the scheduler
105  Scheduler::Schedule(); // run the scheduler - determine the first thread to run
106 
107  Thread_Switch(); // Set the next scheduled thread to the current thread
108 
109  KernelTimer::Start(); // enable the kernel timer
110  KernelSWI::Start(); // enable the task switch SWI
111 
112 #if KERNEL_ROUND_ROBIN
113  // Restart the thread quantum timer, as any value held prior to starting
114  // the kernel will be invalid. This fixes a bug where multiple threads
115  // started with the highest priority before starting the kernel causes problems
116  // until the running thread voluntarily blocks.
117  Quantum::Update(g_pclCurrent);
118 #endif // #if KERNEL_ROUND_ROBIN
119 
120  // Restore the context...
121  Thread_RestoreContext(); // restore the context of the first running thread
122  ASM("reti"); // return from interrupt - will return to the first scheduled thread
123 }
