29 //---------------------------------------------------------------------------
30 Thread::~Thread()
31 {
32  // On destruction of a thread located on a stack,
33  // ensure that the thread is either stopped, or exited.
34  // If the thread is stopped, move it to the exit state.
35  // If not in the exit state, kernel panic -- it's catastrophic to have
36  // running threads on stack suddenly disappear.
37  if (m_eState == ThreadState::Stop) {
38  CS_ENTER();
39  m_pclCurrent->Remove(this);
40  m_pclCurrent = 0;
41  m_pclOwner = 0;
42  m_eState = ThreadState::Exit;
43  CS_EXIT();
44  } else if (m_eState != ThreadState::Exit) {
45  Kernel::Panic(PANIC_RUNNING_THREAD_DESCOPED);
46  }
47 }

110 void Thread::Start(void)
111 {
112  KERNEL_ASSERT(IsInitialized());
113 
114  // Remove the thread from the scheduler's "stopped" list, and add it
115  // to the scheduler's ready list at the proper priority.
116 
117  CS_ENTER();
118  Scheduler::GetStopList()->Remove(this);
119  Scheduler::Add(this);
120  m_pclOwner = Scheduler::GetThreadList(m_uXPriority);
121  m_pclCurrent = m_pclOwner;
122  m_eState = ThreadState::Ready;
123 
124 #if KERNEL_ROUND_ROBIN
125  if (Kernel::IsStarted()) {
126  if (GetCurPriority() >= Scheduler::GetCurrentThread()->GetCurPriority()) {
127  // Deal with the thread Quantum
128  Quantum::Update(this);
129  }
130  }
131 #endif
132 
133  if (Kernel::IsStarted()) {
134  if (GetCurPriority() >= Scheduler::GetCurrentThread()->GetCurPriority()) {
135  Thread::Yield();
136  }
137  }
138  CS_EXIT();
139 }

141 //---------------------------------------------------------------------------
142 void Thread::Stop()
143 {
144  KERNEL_ASSERT(IsInitialized());
145 
146  auto bReschedule = false;
147  if (m_eState == ThreadState::Stop) {
148  return;
149  }
150 
151  CS_ENTER();
152 
153  // If a thread is attempting to stop itself, ensure we call the scheduler
154  if (this == Scheduler::GetCurrentThread()) {
155  bReschedule = true;
156 #if KERNEL_ROUND_ROBIN
157  // Cancel RR scheduling
158  Quantum::Cancel();
159 #endif
160  }
161 
162  // Add this thread to the stop-list (removing it from active scheduling)
163  // Remove the thread from scheduling
164  if (m_eState == ThreadState::Ready) {
165  Scheduler::Remove(this);
166  } else if (m_eState == ThreadState::Blocked) {
167  m_pclCurrent->Remove(this);
168  }
169 
170  m_pclOwner = Scheduler::GetStopList();
171  m_pclCurrent = m_pclOwner;
172  m_pclOwner->Add(this);
173  m_eState = ThreadState::Stop;
174 
175  // Just to be safe - attempt to remove the thread's timer
176  // from the timer-scheduler (does no harm if it isn't
177  // in the timer-list)
178  TimerScheduler::Remove(&m_clTimer);
179 
180  CS_EXIT();
181 
182  if (bReschedule) {
183  Thread::Yield();
184  }
185 }

187 //---------------------------------------------------------------------------
188 void Thread::Exit()
189 {
190  KERNEL_ASSERT(IsInitialized());
191 
192  bool bReschedule = false;
193 
194  if (m_eState == ThreadState::Exit) {
195  return;
196  }
197 
198  CS_ENTER();
199 
200  // If this thread is the actively-running thread, make sure we run the
201  // scheduler again.
202  if (this == Scheduler::GetCurrentThread()) {
203  bReschedule = true;
204 #if KERNEL_ROUND_ROBIN
205  // Cancel RR scheduling
206  Quantum::Cancel();
207 #endif
208  }
209 
210  // Remove the thread from scheduling
211  if (m_eState == ThreadState::Ready) {
212  Scheduler::Remove(this);
213  } else if ((m_eState == ThreadState::Blocked) || (m_eState == ThreadState::Stop)) {
214  m_pclCurrent->Remove(this);
215  }
216 
217  m_pclCurrent = 0;
218  m_pclOwner = 0;
219  m_eState = ThreadState::Exit;
220 
221  // We've removed the thread from scheduling, but interrupts might
222  // trigger checks against this thread's currently priority before
223  // we get around to scheduling new threads. As a result, set the
224  // priority to idle to ensure that we always wind up scheduling
225  // new threads.
226  m_uXCurPriority = 0;
227  m_uXPriority = 0;
228 
229  // Just to be safe - attempt to remove the thread's timer
230  // from the timer-scheduler (does no harm if it isn't
231  // in the timer-list)
232  TimerScheduler::Remove(&m_clTimer);
233  CS_EXIT();
234 
235 #if KERNEL_THREAD_EXIT_CALLOUT
236  ThreadExitCallout pfCallout = Kernel::GetThreadExitCallout();
237  if (pfCallout != nullptr) {
238  pfCallout(this);
239  }
240 #endif
241 
242  if (bReschedule) {
243  // Choose a new "next" thread if we must
244  Thread::Yield();
245  }
246 }

248 //---------------------------------------------------------------------------
249 void Thread::Sleep(uint32_t u32TimeMs_)
250 {
251  Semaphore clSemaphore;
252  auto* pclTimer = g_pclCurrent->GetTimer();
253  auto lTimerCallback = [](Thread* /*pclOwner*/, void* pvData_) {
254  auto* pclSemaphore = static_cast<Semaphore*>(pvData_);
255  pclSemaphore->Post();
256  };
257 
258  // Create a semaphore that this thread will block on
259  clSemaphore.Init(0, 1);
260 
261  // Create a one-shot timer that will call a callback that posts the
262  // semaphore, waking our thread.
263  pclTimer->Init();
264  pclTimer->Start(false, u32TimeMs_, lTimerCallback, &clSemaphore);
265 
266  clSemaphore.Pend();
267 }

271 uint16_t Thread::GetStackSlack()
272 {
273  KERNEL_ASSERT(IsInitialized());
274 
275  K_ADDR wBottom = 0;
276  auto wTop = static_cast<K_ADDR>(m_u16StackSize - 1);
277  auto wMid = ((wTop + wBottom) + 1) / 2;
278 
279  CS_ENTER();
280 
281  // Logarithmic bisection - find the point where the contents of the
282  // stack go from 0xFF's to non 0xFF. Not Definitive, but accurate enough
283  while ((wTop - wBottom) > 1) {
284 #if STACK_GROWS_DOWN
285  if (m_pwStack[wMid] != (K_WORD)(-1))
286 #else
287  if (m_pwStack[wMid] == (K_WORD)(-1))
288 #endif
289  {
291  wTop = wMid;
292  } else {
293  wBottom = wMid;
294  }
295  wMid = (wTop + wBottom + 1) / 2;
296  }
297 
298  CS_EXIT();
299 
300  return wMid;
301 }

305 void Thread::Yield()
306 {
307  CS_ENTER();
308  // Run the scheduler
309  if (Scheduler::IsEnabled()) {
310  Scheduler::Schedule();
311 
312  // Only switch contexts if the new task is different than the old task
313  if (g_pclCurrent != g_pclNext) {
314 #if KERNEL_ROUND_ROBIN

316 #endif
317  Thread::ContextSwitchSWI();
318  }
319  } else {
320  Scheduler::QueueScheduler();
321  }

344 {
345  KERNEL_ASSERT(IsInitialized());
346  auto bSchedule = false;
347 
348  CS_ENTER();
349 
350  // If this is the currently running thread, it's a good idea to reschedule
